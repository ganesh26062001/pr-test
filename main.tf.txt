# Configure the AWS Provider
provider "aws" {
  region = "us-east-1"
}

# Define a resource block to create an AWS EC2 instance
resource "aws_instance" "web_server" {
  # Specify the Amazon Machine Image (AMI) ID
  # This AMI ID (ami-0261755bbcb8c4a84) is for Ubuntu Server 18.04 LTS in us-east-1
  ami           = "ami-0261755bbcb8c4a84"

  # Specify the instance type (e.g., t2.micro, often in the free tier)
  instance_type = "t2.micro"

  # Optional: Define tags for the instance
  tags = {
    Name = "TerraformWebServer"
  }
}
